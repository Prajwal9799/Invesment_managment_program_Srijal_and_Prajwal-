package application;

import java.util.ArrayList;
import javafx.application.Application;
import javafx.collections.FXCollections;
import javafx.collections.ObservableList;
import javafx.event.ActionEvent;
import javafx.event.EventHandler;
import javafx.geometry.Insets;
import javafx.geometry.Pos;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.CheckBox;
import javafx.scene.control.ChoiceBox;
import javafx.scene.control.DatePicker;
import javafx.scene.control.Label;
import javafx.scene.control.ListView;
import javafx.scene.control.RadioButton;
import javafx.scene.control.TextField;
import javafx.scene.control.ToggleGroup;

import javafx.scene.layout.GridPane;
import javafx.stage.Stage;

import javafx.scene.paint.Color;
import javafx.scene.shape.Circle;
import javafx.scene.Group;

public class trade extends Application {

	public static void main(String[] args) {
		// TODO Auto-generated method stub
		launch(args);

	}

	@Override
	public void start(Stage primaryStage) throws Exception {

		primaryStage.setTitle("Trading Platform");

		// TODO Auto-generated method stub
		GridPane myGrid = new GridPane();

		myGrid.setMinSize(400, 200);

		myGrid.setMinSize(500, 500);
		myGrid.setPadding(new Insets(10, 10, 10, 10));
		myGrid.setVgap(15);
		myGrid.setHgap(15);

		Label id = new Label("exchangeID: ");
		Label stocksAvailable = new Label("Stock");
		Label purchaseDate = new Label("Date");
		Label stocksBought = new Label("Number of Shares");
		Label payment = new Label("Enter Payment Method");

		TextField text = new TextField();
		DatePicker date = new DatePicker();
		// implementing toggle
		final ToggleGroup group = new ToggleGroup();

		// checkBoxes
		CheckBox javaOption = new CheckBox("Java");
		CheckBox dotnetOption = new CheckBox("Dot Net");

		String paymentAvailableOptions[] = { "Account Balance", "Credit", "Debit" };
		ChoiceBox mypaymentAvailable = new ChoiceBox(FXCollections.observableArrayList(paymentAvailableOptions));

		// Choice of Stocks
		String stocksAvailableOptions[] = { "TSLA", "APL", "HP", "META", "AMZ", "YMT", "XLM" };
		ChoiceBox mystocksAvailable = new ChoiceBox(FXCollections.observableArrayList(stocksAvailableOptions));

		// int sharesAvailableOptions[] = {1, 2, 3, 4, 5};
		String sharesAvailableOptions[] = { "1", "2", "3", "4", "5" };
		ChoiceBox mysharesAvailable = new ChoiceBox(FXCollections.observableArrayList(sharesAvailableOptions));

		Button purchase = new Button("Purchase");

		myGrid.addRow(5, payment, mypaymentAvailable);
		myGrid.addRow(0, id, text);
		myGrid.addRow(1, stocksAvailable, mystocksAvailable);
		myGrid.addRow(2, purchaseDate, date);
		myGrid.addRow(3, stocksBought, mysharesAvailable);

		myGrid.add(purchase, 4, 9);

		Scene scene = new Scene(myGrid);
		primaryStage.setScene(scene);
		primaryStage.show();

		purchase.setOnAction((EventHandler<ActionEvent>) new EventHandler<ActionEvent>() {
			@Override
			public void handle(ActionEvent e) {
				GridPane myGrid1 = new GridPane();
				myGrid1.setMinSize(400, 200);
				Label registration = new Label("Your purchase has been made successfully.");
				myGrid1.setAlignment(Pos.CENTER);
				myGrid1.addRow(2, registration);
				Stage s = new Stage();
				Scene scene1 = new Scene(myGrid1);
				s.setScene(scene1);
				s.show();
			}
		});

	}
}
